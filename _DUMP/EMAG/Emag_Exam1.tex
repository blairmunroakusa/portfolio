\begin{IEEEeqnarray*}{lrCl}
\comm{E-field}&\mathbf{E}(z)&=&\hat x E_0 e^{-\alpha z}e^{-j\beta z}\\
\comm{H-field}&\mathbf{H}(z)&=&\frac{1}{\eta}\hat z \times \mathbf{E}\\
